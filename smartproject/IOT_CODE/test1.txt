import RPi.GPIO as GPIO
import time
import os

# GPIO setup
GPIO.setmode(GPIO.BCM)

# Ultrasonic sensor pins
TRIG = 23  # Trigger pin
ECHO = 24  # Echo pin

GPIO.setup(TRIG, GPIO.OUT)
GPIO.setup(ECHO, GPIO.IN)

# MP3 file and cooldown settings
MP3_FILE = "alert.mp3"
DISTANCE_THRESHOLD = 20  # cm
COOLDOWN = 60  # seconds

def get_distance():
    """Measure distance using ultrasonic sensor."""
    # Send 10us pulse to trigger
    GPIO.output(TRIG, True)
    time.sleep(0.00001)
    GPIO.output(TRIG, False)

    # Wait for echo start
    while GPIO.input(ECHO) == 0:
        pulse_start = time.time()
    
    # Wait for echo end
    while GPIO.input(ECHO) == 1:
        pulse_end = time.time()
    
    # Calculate distance (speed of sound = 34300 cm/s)
    pulse_duration = pulse_end - pulse_start
    distance = pulse_duration * 17150  # Distance = (time * speed) / 2
    return round(distance, 2)

def play_mp3():
    """Play the MP3 file using mpg123."""
    os.system(f"mpg123 {MP3_FILE} &")  # Run in background

def main():
    last_trigger_time = 0
    print("Ultrasonic Sensor MP3 Trigger Started...")

    try:
        while True:
            distance = get_distance()
            current_time = time.time()

            # Check if object is within threshold and cooldown has passed
            if distance < DISTANCE_THRESHOLD and (current_time - last_trigger_time) > COOLDOWN:
                print(f"Object detected at {distance} cm! Playing MP3...")
                play_mp3()
                last_trigger_time = current_time
            elif distance < DISTANCE_THRESHOLD:
                print(f"Object detected at {distance} cm, but in cooldown...")
            else:
                print(f"No object detected. Distance: {distance} cm")

            time.sleep(0.5)  # Check every 0.5 seconds

    except KeyboardInterrupt:
        print("\nProgram stopped.")
    finally:
        GPIO.cleanup()
        print("GPIO cleanup complete.")

if __name__ == "__main__":
    # Ensure trigger is low initially
    GPIO.output(TRIG, False)
    time.sleep(2)  # Allow sensor to settle
    main()
